\begin{definition}[Magic argument check]
  \begin{flalign*}
    \osFunc{CheckArgs}(\ovalue, n) =
    \begin{cases}
      \ostrue, & \text{if } (\ovalue, n) =
        \begin{cases}
          (\osMagic{Type}, 1) \\
          (\osMagic{IntAdd}, 2) \\
          (\osMagic{IntNeg}, 1) \\
          (\osMagic{StrAdd}, 2) \\
          (\osMagic{StrContains}, 2) \\
          (\osMagic{LstGetitem}, 2) \\
          (\osMagic{TplGetitem}, 2) \\
        \end{cases}
      \\
      \osfalse, & \text{otherwise}
    \end{cases}
  \end{flalign*}
\end{definition}
%</helper>
